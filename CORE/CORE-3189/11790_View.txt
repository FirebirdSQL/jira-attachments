1) This view functions in all the versions of the Firebird, except in Firebird 2.5:

CREATE OR ALTER VIEW SALDO(
    MES_ANO,
    CODIGO_SETOR,
    CODIGO_PRODUTO,
    LOTE,
    SLD_INICIO,
    SLD_ENTRADA,
    SLD_SAIDA,
    SLD_CONSUMO,
    EXCLUIR)
AS
select PD.MES_ANO, PD.CODIGO_SETOR, PD.CODIGO_PRODUTO, PD.LOTE, (select SUM(PDI.QTD_INICIO) from PRATELEIRA_DIVISORIA PDI  where PDI.CODIGO_PRODUTO=PD.CODIGO_PRODUTO and PDI.CODIGO_SETOR=PD.CODIGO_SETOR and PDI.LOTE=PD.LOTE and PDI.MES_ANO=PD.MES_ANO) AS QTD_INICIO, (select SUM(PDI.QTD_ENTRADA) from PRATELEIRA_DIVISORIA PDI  where PDI.CODIGO_PRODUTO=PD.CODIGO_PRODUTO and PDI.CODIGO_SETOR=PD.CODIGO_SETOR and PDI.LOTE=PD.LOTE and PDI.MES_ANO=PD.MES_ANO) AS QTD_ENTRADA, (select SUM(PDI.QTD_SAIDA) from PRATELEIRA_DIVISORIA PDI  where PDI.CODIGO_PRODUTO=PD.CODIGO_PRODUTO and PDI.CODIGO_SETOR=PD.CODIGO_SETOR and PDI.LOTE=PD.LOTE and PDI.MES_ANO=PD.MES_ANO) AS QTD_SAIDA, (select SUM(PDI.QTD_CONSUMO) from PRATELEIRA_DIVISORIA PDI  where PDI.CODIGO_PRODUTO=PD.CODIGO_PRODUTO and PDI.CODIGO_SETOR=PD.CODIGO_SETOR and PDI.LOTE=PD.LOTE and PDI.MES_ANO=PD.MES_ANO) AS QTD_CONSUMO, PD.EXCLUIR
  from PRATELEIRA_DIVISORIA PD
  inner join PRATELEIRA_DIVISORIA PDI on PD.MES_ANO = PDI.MES_ANO and PD.CODIGO_SETOR = PDI.CODIGO_SETOR and PD.CODIGO_PRODUTO = PDI.CODIGO_PRODUTO and PD.LOTE = PDI.LOTE
  where PD.EXCLUIR<>'S'
  group by PD.MES_ANO, PD.CODIGO_SETOR, PD.CODIGO_PRODUTO, PD.LOTE, PD.EXCLUIR
  order by PD.MES_ANO, PD.CODIGO_SETOR, PD.CODIGO_PRODUTO, PD.LOTE
;




2) Modified View to function in Firebird 2.5:

CREATE OR ALTER VIEW SALDO(
    MES_ANO,
    CODIGO_SETOR,
    CODIGO_PRODUTO,
    LOTE,
    SLD_INICIO,
    SLD_ENTRADA,
    SLD_SAIDA,
    SLD_CONSUMO,
    EXCLUIR)
AS
with CTE_SALDO (MES_ANO, CODIGO_SETOR, CODIGO_PRODUTO, LOTE, SLD_INICIO, SLD_ENTRADA, SLD_SAIDA, SLD_CONSUMO, EXCLUIR)
as (select PD.MES_ANO, PD.CODIGO_SETOR, PD.CODIGO_PRODUTO, PD.LOTE,
           (select sum(PDI.QTD_INICIO)
            from PRATELEIRA_DIVISORIA PDI
            where PDI.CODIGO_PRODUTO = PD.CODIGO_PRODUTO and
                  PDI.CODIGO_SETOR = PD.CODIGO_SETOR and
                  PDI.LOTE = PD.LOTE and
                  PDI.MES_ANO = PD.MES_ANO) as QTD_INICIO,
           (select sum(PDI.QTD_ENTRADA)
            from PRATELEIRA_DIVISORIA PDI
            where PDI.CODIGO_PRODUTO = PD.CODIGO_PRODUTO and
                  PDI.CODIGO_SETOR = PD.CODIGO_SETOR and
                  PDI.LOTE = PD.LOTE and
                  PDI.MES_ANO = PD.MES_ANO) as QTD_ENTRADA,
           (select sum(PDI.QTD_SAIDA)
            from PRATELEIRA_DIVISORIA PDI
            where PDI.CODIGO_PRODUTO = PD.CODIGO_PRODUTO and
                  PDI.CODIGO_SETOR = PD.CODIGO_SETOR and
                  PDI.LOTE = PD.LOTE and
                  PDI.MES_ANO = PD.MES_ANO) as QTD_SAIDA,
           (select sum(PDI.QTD_CONSUMO)
            from PRATELEIRA_DIVISORIA PDI
            where PDI.CODIGO_PRODUTO = PD.CODIGO_PRODUTO and
                  PDI.CODIGO_SETOR = PD.CODIGO_SETOR and
                  PDI.LOTE = PD.LOTE and
                  PDI.MES_ANO = PD.MES_ANO) as QTD_CONSUMO, PD.EXCLUIR
    from PRATELEIRA_DIVISORIA PD
    inner join PRATELEIRA_DIVISORIA PDI on PD.MES_ANO = PDI.MES_ANO and PD.CODIGO_SETOR = PDI.CODIGO_SETOR and PD.CODIGO_PRODUTO = PDI.CODIGO_PRODUTO and PD.LOTE = PDI.LOTE
    where PD.EXCLUIR <> 'S'
    group by PD.MES_ANO, PD.CODIGO_SETOR, PD.CODIGO_PRODUTO, PD.LOTE, PD.EXCLUIR
    order by PD.MES_ANO, PD.CODIGO_SETOR, PD.CODIGO_PRODUTO, PD.LOTE)
select MES_ANO, CODIGO_SETOR, CODIGO_PRODUTO, LOTE, coalesce(SLD_INICIO, 0) as QTD_INICIO,
       coalesce(SLD_ENTRADA, 0) as QTD_ENTRADA, coalesce(SLD_SAIDA, 0) as QTD_SAIDA,
       coalesce(SLD_CONSUMO, 0) as QTD_CONSUMO, EXCLUIR
from CTE_SALDO
;

select S.CODIGO_PRODUTO, P.NOME AS PRODUTO, P.CODIGO_GRUPO_PRODUTO, GP.NOME AS GRUPO_PRODUTO, S.CODIGO_SETOR, ST.NOME AS SETOR,
       cast((select sum(SLD_INICIO+SLD_ENTRADA-SLD_SAIDA) from SALDO where CODIGO_PRODUTO=S.CODIGO_PRODUTO and CODIGO_SETOR=S.CODIGO_SETOR and MES_ANO=S.MES_ANO and EXCLUIR<>'S' and (SLD_INICIO+SLD_ENTRADA-SLD_SAIDA)>0)as numeric(12,4)) as ESTOQUE,
       S.LOTE, L.VALIDADE, PD.PRATELEIRA, PD.DIVISORIA, cast((PD.QTD_INICIO+PD.QTD_ENTRADA-PD.QTD_SAIDA) as numeric(12,4)) AS QUANTIDADE, UM.ABREVIACAO
       from SALDO S
       inner join PRODUTO P on S.CODIGO_PRODUTO = P.CODIGO_PRODUTO
       inner join GRUPO_PRODUTO GP on P.CODIGO_GRUPO_PRODUTO = GP.CODIGO_GRUPO_PRODUTO
       inner join SETOR ST on S.CODIGO_SETOR = ST.CODIGO_SETOR
       inner join LOTE L on S.LOTE = L.LOTE and S.CODIGO_PRODUTO = L.CODIGO_PRODUTO
       inner join UNIDADE_MEDIDA UM on P.CODIGO_UNIDADE_MEDIDA = UM.CODIGO_UNIDADE_MEDIDA
       left join PRATELEIRA_DIVISORIA PD on S.LOTE = PD.LOTE and S.CODIGO_SETOR = PD.CODIGO_SETOR AND S.CODIGO_PRODUTO = PD.CODIGO_PRODUTO and S.MES_ANO=PD.MES_ANO
       where S.EXCLUIR<>'S' and (PD.QTD_INICIO+PD.QTD_ENTRADA-PD.QTD_SAIDA)>0 and S.mes_ano='01/2010' and p.codigo_grupo_produto ='19' and (S.SLD_INICIO+S.SLD_ENTRADA-S.SLD_SAIDA)>0
       order by ST.NOME, GP.NOME, P.NOME, L.LOTE, PD.PRATELEIRA, PD.DIVISORIA
